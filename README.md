# Comandos para la instalación
- apt update && apt-get upgrade -y
- apt install git
- git clone https://github.com/n4ss4u/ScrapNow
- apt install python3 -y 
- cd ScrapNow
- pip install -r requirements.txt
- python3 scrapnow.py example.com
